Calcul des températures suivantes sur 5 incréments de temps¶

levels=np.linspace(0,70,8) #On subdivise en 7 intervalles (avec 8 bornes) le domaine de temprature ([0;70C])

for n in range(1,6): #Pour 5 intervalles de temps
    
    # On calcule la temprature en chaque noeud (i,j) en fonction de celle  l'instant prcdent
    for i in range(1,10):
        for j in range(1,10):
            T[i,j]=T[i,j] + h/k**2 * (T[i-1,j]+T[i+1,j]+T[i,j-1]+T[i,j+1]-4*T[i,j]) # Nouveau T[i,j] en fonction de l'ancien
    
    #On trace un graphique en contours
    cont=plt.contourf(np.array(T),levels=levels);plt.colorbar(cont);
    plt.savefig('Im_profils_temper/fig'+str(n)+'.png')
    plt.show()
    
    # On rcupre  chaque itration la temprature au noeud (5,5)
    listeT55.append(T[5,5])
